# SI Prefix Test

1k                                      # => 1,000
1M                                      # => 1,000,000
1G                                      # => 1,000,000,000
1T                                      # => 1,000,000,000,000
1P                                      # => 1,000,000,000,000,000

2.5k + 500                              # => 3,000
10G / 4                                 # => 2,500,000,000

# Small prefixes

1n                                      # => 0.000000001
1p                                      # => 0.000000000001
1f                                      # => 0.000000000000001

500n * 2                                # => 0.000001
100u + 50u                              # => 0.00015
